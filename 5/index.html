La boucle "for"

}

La condition

var toto = 'toto';
var tata = 'tata';
var toto2 = 'toto';

if(toto == tata){
	console.log('Oops internet est cassé !');
}

if(toto2 == toto){
	console.log('Internet se porte bien, vous pouvez continuer votre navigation');
}
